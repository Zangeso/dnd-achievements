document.getElementById('submitBtn').onclick = async () => { const title = document.getElementById('title').value; const select = document.getElementById('playerSelect'); const playerId = select.value; const playerName = select.options[select.selectedIndex].text; const playerAvatar = select.options[select.selectedIndex].getAttribute('data-avatar') || ''; const file = document.getElementById('imageFile').files[0]; if (!title || !file) { alert('Заполните название и выберите картинку'); return; } try { console.log('Начинаем загрузку картинки...'); const imageUrl = await uploadImage(file); console.log('Картинка загружена, URL:', imageUrl); const insertData = { title, player_name: playerName, player_vk_id: parseInt(playerId), player_avatar: playerAvatar, image_url: imageUrl }; console.log('Вставляем в таблицу:', insertData); const { data, error } = await supabaseClient .from('achievements') .insert([insertData]) .select(); if (error) { console.error('Ошибка вставки:', error); alert('Ошибка базы данных: ' + error.message); return; } console.log('Успешно вставлено:', data); document.getElementById('message').innerText = '✅ Достижение опубликовано!'; document.getElementById('title').value = ''; document.getElementById('imageFile').value = ''; } catch (err) { console.error('Общая ошибка:', err); alert('Ошибка: ' + (err.message || err)); } };
